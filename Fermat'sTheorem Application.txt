 n = Double.parseDouble(jTextField2.getText());
         switch (day) {
            case "sunday":

                jLabel3.setVisible(false);
                break;
            case "monday":
                a=1;

            case "tuesday":

                case "SUNDAY":

            case "MONDAY":

            case "TUESDAY":

            case "THURSDAY":

                break;
            case "FRIDAY":
                a=5;
                jLabel3.setVisible(false);
                break;
            case "SATAURDAY":
                a=6;
                jLabel3.setVisible(false);
                break;
            default:
                jLabel3.setVisible(true);
                break;
        }
        
if(jRadioButton1.isSelected()==true)
 {
    p = Double.parseDouble(jTextField3.getText()); 
    nm=n%7;
pm=p%6;
        if(nm==0 && pm==0)
            {
                    b=0;
                    jLabel3.setVisible(false);
            }
        else
        {
            b=Math.pow(nm, pm);
        }
    c=b%7;
    d=c+a;
if(d>=7)
{
    d=d%7;
}
 }
else if(jRadioButton1.isSelected()==false)
{
    jTextField3.setText("1");
    nm=n%7;
pm=1;
b=Math.pow(nm, pm);
c=b%7;
d=c+a;
if(d>=7)
{
    d=d%7;
}
}

if(d==0)
{
    jTextField4.setText("Sunday");
}
else if(d==1)
{
    jTextField4.setText("monday");
}
else if(d==2)
{
    jTextField4.setText("tuesday");
}
else if(d==3)
{
    jTextField4.setText("wednesday");
}
else if(d==4)
{
    jTextField4.setText("thursday");
}
else if(d==5)
{
    jTextField4.setText("friday");
}
else if(d==6)
{
    jTextField4.setText("sataurday");
}
 
    }
